## safir 的 click 事件监听

```js
var a = document.createElement('a');
a.download = saveName;
var objURL = URL.createObjectURL(saveBlob);
a.href = objURL;
try {
    var ev = new MouseEvent('click', {
        "view": window,
        "bubbles": true,
        "cancelable": false
    });
    a.dispatchEvent(ev);
} catch (ex) {
    try { a.click(); } catch (ex) { }
}
setTimeout(function () {
    URL.revokeObjectURL(objURL);
}, 10000);
```
